```julia
function run_two_stage_optimization(problem, initial_params, loss_config, opt_config)
    # Stage 1: LBFGS
    res1 = solve(problem, OptimizationOptimJL.LBFGS(); maxiters=50)
    
    # Stage 2: Adam с scheduler
    res2 = solve(problem, OptimizationOptimisers.Adam(lr); maxiters=2950)
    
    return (stage1=res1, stage2=res2, final=res2.u)
end
```
